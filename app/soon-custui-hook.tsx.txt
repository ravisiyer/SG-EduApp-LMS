'use client';
import React, { useState } from 'react';
import { View, Text, ActivityIndicator, TextInput, Button, Alert, Platform } from 'react-native';
import '@/global.css';
import { useSignUp } from '@clerk/clerk-expo';
import { useRouter } from 'expo-router';

let WaitlistComponent: React.ComponentType<any> | null = null;

if (Platform.OS === 'web') {
  // Web-only Waitlist
  const { Waitlist } = require('@clerk/clerk-react');
  WaitlistComponent = Waitlist;
}

const Page = () => {
  const router = useRouter();

  // Mobile native state
  // const { signUp, setSession, isLoaded } = useSignUp();
  const { signUp, isLoaded } = useSignUp();
  const [email, setEmail] = useState('');
  const [loading, setLoading] = useState(false);

  const handleNativeWaitlist = async () => {
    if (!email) return Alert.alert('Please enter your email');
    if (!signUp) return;

    try {
      setLoading(true);
      // Create a new sign up attempt (waitlist)
      await signUp.create({ emailAddress: email });
      Alert.alert('Success', 'You have joined the waitlist!');
      router.push('/wait'); // redirect to wait page
    } catch (err: any) {
      console.error(err);
      Alert.alert('Error', err?.message || 'Failed to join waitlist');
    } finally {
      setLoading(false);
    }
  };

  return (
    <View className="flex-1 items-center justify-center bg-white dark:bg-black p-4">
      <View className="mb-10 justify-center items-center">
        <Text className="text-4xl font-bold text-center">
          We're not yet open to the public
        </Text>
        <Text className="text-sm text-gray-500 text-center mt-2">
          Please join the waitlist to get early access to the app
        </Text>
      </View>

      {Platform.OS === 'web' && WaitlistComponent ? (
        <WaitlistComponent
          appearance={{ variables: { colorPrimary: '#0d6c9a' } }}
          afterJoinWaitlistUrl="/wait"
          signInUrl="/login"
          fallback={<ActivityIndicator />}
        />
      ) : (
        // Mobile native waitlist form
        <View className="w-full max-w-md">
          <TextInput
            placeholder="Enter your email"
            value={email}
            onChangeText={setEmail}
            keyboardType="email-address"
            autoCapitalize="none"
            className="border border-gray-300 rounded px-4 py-2 mb-4 text-black dark:text-white"
          />
          <Button
            title={loading ? 'Joining...' : 'Join Waitlist'}
            onPress={handleNativeWaitlist}
            disabled={loading || !isLoaded || !signUp}
          />
          {loading && <ActivityIndicator className="mt-2" />}
        </View>
      )}
    </View>
  );
};

export default Page;
